% FSD Injection
\subsection{FSD Injection (буст прыжка)}
\eddim{Используются только Raw-материалы}

\begin{tcolorbox}[edbox-blue, title=BASIC +25\%]
\begin{itemize}
\item Carbon (Углерод)
\item Vanadium (Ванадий)
\item Germanium (Германий)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}[edbox-blue, title=STANDARD +50\%]
\begin{itemize}
\item Carbon / Vanadium / Germanium
\item Cadmium (Кадмий)
\item Niobium (Ниобий)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}[edbox-blue, title=PREMIUM +100\%]
\begin{itemize}
\item Carbon / Vanadium / Germanium
\item Arsenic (Мышьяк)
\item Yttrium (Иттрий)
\item Polonium (Полоний)
\end{itemize}
\end{tcolorbox}
